Depth First Search(DFS) – Use Stack

2010-12-24
  • Algorithms
    • Programming
  • Algorithms
  • DFS
  • Stack

This post implements a simple DFS based on stack theory, the following code snippet accept a graph as input, and validate whether it is a connected graph.



#include <iostream>


#include <list>


#include <stack>


using namespace std;


void DFS(int,list<int>,bool);




int main(){


    int node,edge,j,a,b;


    list<int> rList;


    bool isVisited;


    while(cin>>node&&node!=0){


        rList = new list<int>[node+1];


        isVisited = new bool[node+1];


        isVisited[0] = true;


        for(j=1;j<=node;j++)


            isVisited[j]=false;




        cin>>edge;


        for(j=0;j<edge;j++){


            cin>>a>>b;


            rList[a].push_back(b);


            rList[b].push_back(a);


        }


        DFS(1,rList,isVisited);


    }


    return 0;


}


void DFS(int node,list<int> rList,bool isVisited){


    bool haveNode;


    int a;


    int lastNode = new int[node+1];


    for(int j=0;j<=node;j++) lastNode[j]=1;




    list<int>::iterator i,iend;


    stack<int> s;


    s.push(node);


    cout<<“A DFS path is:”<<endl;


    while(!s.empty()){


        a = s.top();


        s.pop();


        isVisited[a]=true;


        cout<<“Search to:”<<a<<endl;


        iend = rList[a].end();


        haveNode = false;


        for(i=rList[a].begin();i!=iend;i++){


            if(!isVisited[i]){


                s.push(i);


                lastNode[i]=a;


                haveNode = true;


            }


        }


        if(!haveNode){


            s.push(lastNode[a]);


            if(lastNode[a]==1){


                bool flag = false;


                iend = rList[1].end();


                for(i=rList[1].begin();i!=iend;i++){


                    if(!isVisited[*i])


                        flag = true;


                }


                if(!flag){


                    cout<<“Back to 1”<<endl;


                    break;


                }


                s.push(1);


            }


        }


    }


    int ct = 0;


    for(int j=1;j<=node;j++){


        if(isVisited[j]) ct++;


    }


    if(ct == node) cout<<“Is a Map”<<endl;


    else cout<<“Is not a Map”<<endl;


}




Depth First Search(DFS) – Use Recursion

2010-10-24

This post implements a simple DFS based on recursive theory, the following code snippet accept a graph as input, and validate whether it is a connected graph.



#include <iostream>


#include <list>


using namespace std;




bool DFS(int,list<int>,bool);


void setFlag(int,list<int>,bool,int);




bool DFS(int node,list<int> rList,bool isVisited){


    int lastNode = new int[node+1];


    for(int j=0;j<=node;j++)


        lastNode[j]=0;


    for(int j=1;j<=node;j++)


        isVisited[j]=false;


    cout<<“DFS path is :”<<endl;


    setFlag(1,rList,isVisited,lastNode);


}


void setFlag(int node,list<int> rList,bool isVisited,int lastNode){


    bool haveNode = false;


    list<int>::iterator i,iend;


    isVisited[node] = true;


    if(lastNode[node]!=0)


        cout<<lastNode[node]<<“-“<<node<<endl;


    iend = rList[node].end();


    for(i=rList[node].begin();i!=iend;i++){


        if(!isVisited[i]){


            lastNode[i] = node;


            setFlag(i,rList,isVisited,lastNode);


            haveNode = true;


        }


    }


    if(!haveNode){


        if(lastNode[node]==1){


            bool flag = false;


            iend = rList[1].end();


            for(i=rList[1].begin();i!=iend;i++){


                if(!isVisited[i])


                    flag = true;


            }


            if(!flag)    return;


        }


        setFlag(lastNode[node],rList,isVisited,lastNode);


    }


}




int main(){


    int node,edge,j,a,b,ct;


    list<int> rList;


    bool *isVisited;


    while( cin>>node&&node!=0){


        rList = new list<int>[node+1];


        isVisited = new bool[node+1];


        isVisited[0] = true;




        cin>>edge;


        for(j=0;j<edge;j++){


            cin>>a>>b;


            rList[a].push_back(b);


            rList[b].push_back(a);


        }


        DFS(node,rList,isVisited);


        ct = 0;


        for(j=1;j<=node;j++){


            if(isVisited[j]) ct++;


        }


        if(ct == node) cout<<“Is a Map”<<endl;


        else cout<<“Is not a Map”<<endl;


    }


    return 0;


}




Disappears

2010-05-24
  • Kevin's Life
  • Freelance
  • My Life

人活着就会思考，追求，生活也会有寄托。
最近一个月，感觉自己像是人间蒸发掉了一样，不上QQ，不发短信，也不打电话，所有的联络都被自己主动的掐断。而在这样封闭的生活里，一个月的时间瞬间即逝。我的改变让家人有点担心了，心里不想让他们知道我的难处，可是也不希望对他们说谎话，就这样主动的减少一些联络，甚至冷淡一些。

让一个人完完全全蒸发是不可能的，人活着就会思考，追求，生活也会有寄托，这样无声无息的一个人生活实在难受，内心从未像现在这样渴望与人说话。五一节的时候回了合肥，顺道回家呆了两天，因为姐姐结婚的缘故，我热热闹闹的折腾了一番，帮前帮后，然后回家陪父母呆了两天，之后就又回来继续一个人的生活。

从四月底到五一前的那一个星期，我基本是倒时差的过的，没有了工作就没有了固定的时间表，困到不行才睡，加上原本要做的外包软件需要赶工，生物钟十分混乱。五月五号回上海后，正常的过了几天，之后又混乱起来，晚上无聊就看看爱情动作片，之后再无聊就看综艺、韩剧，就这样通过狭小的网络打发时间；有时候一天睡四个小时，有时候睡十六个小时，有时候懒的连出去吃饭都不愿意，有时候又吃零食吃到撑；我从未这样放纵过，一个人的世界里，没有约束，没有顾虑，没有快乐，当然也没有朋友分享心情。

前天晚上去老胡那边吃饭，我们几个人自己买菜做饭，一个人一瓶啤酒，我吃好喝完之后躺了一会；可能是睡眠不足，又吃了好多米饭，最后还是吐了。之后11:20左右坐977回来，竟然又没有了困意，躺在床上，一直看网络剧到第二天中午。

几个星期前就开始上网找外包项目，特别是在国外的rentacoder上转悠，我的感受很明显，想透过写外包软件做自由职业者，是很难维持收入的。我通常把自己对创业的想法扼杀掉，特别是在目前这种封闭式的生活下，更加没有可能性。

我想我是在完全没有整理思绪的情况下写下这篇文章的，每次写这种流水账文章都有着几乎相同的原因，那就是心里有某种想法在滋生，或者是原本平静的心情被起了波澜。这一次，我在思考的是，当我手上唯一的外包项目做完我对工作就没有了寄托，当我在这一个月体验完所有的填补空虚的方法后，接下来我该怎么办。那时的生活会不会变得更加的难以忍受？或者我可以很快找到新的目标？

About My Name

回忆让我拾起起很多零散的片段，在这里拼凑在一起。
最近在校内上看到了一篇文章“中国姓氏英文翻译”，才知道我的姓“奚”翻译成英语通常是Hsi或者Chi，而不是拼音Xi，那这么说我的英文名就应该是Kevin Hsi/Chi了。

“奚”是一个疑问词，可以意译为What/Why/Where/How，而“昊”则是一个名称，可意译为Clear Summer Sky，也有宽广、广大的意思。合起来，我的名字大致就是Wish a clear summer sky，也就是对晴天的期待，就像晴天娃娃(Teru-Teru-Bouzu)一样。

多想用Teru-Teru-Bouzu做我WP博客的名字，可惜不巧有人捷足先登；下面罗列一下我用过的几个网名：

  • 关于“haozi”：高中的时候，朋友们给我起的外号：“昊子”，人生第一个外号，第一个不雅的外号。
  • 关于“cose_haozi”：大二的时候，跟风把学院的缩写加在网名前面(CoSE, College of Software Engineering)。
  • 关于“googledream”：大三的时候，很迷茫，一直在寻找自己的梦想(google是“寻找”的代名词)。
  • 关于“kevin”：第一份工作是在美资外企，我给自己选了一个很常见的英文名字，敲定不变的。
