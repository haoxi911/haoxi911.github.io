Josephus Problem

2010-12-25

问题描述：n个人（编号0~(n-1))，从0开始报数，报到(m-1)的退出，剩下的人继续从0开始报数。求胜利者的编号。

我们知道第一个人(编号一定是m%n-1) 出列之后，剩下的n-1个人组成了一个新的约瑟夫环（以编号为k=m%n的人开始）：k k+1 … n-2, n-1, 0, 1, 2, … k-2，并且从k开始报0。

现在我们把他们的编号做一下转换：
k –> 0
k+1 –> 1
k+2 –> 2
…
k-2 –> n-2
k-1 –> n-1

变换后就完完全全成为了(n-1)个人报数的子问题，假如我们知道这个子问题的解：例如x是最终的胜利者，那么根据上面这个表把这个x变回去：x=(x+k)%n。

递推公式：令f[i]表示i个人玩游戏报m退出最后胜利者的编号，最后的结果自然是f[n]：

f[1]=0;
f[i]=(f[i-1]+m)%i; (i>1)



#include <stdio.h>




int main()


{


  int n, m, i, s=0;


  scanf(“%d%d”, &n, &m);


  for (i=2; i<=n; i++) s=(s+m)%i;


  printf (“The winner is %d\n”, s+1);


  return 0;


}

int fastFind(string &cur,string &pat,int *fail){

    while(posP<lengthP && posT<lengthT){



        if(pat.at(posP)==cur.at(posT)){



            posP++;posT++;//匹配成功一位就同时向后移一位



        }



        else if(posP==0) posT++;//当匹配串没有找到第一个字符时，被匹配字符串向后移

    }



    if(posP<lengthP) return -1;

}

#include <iostream>


#include <list>


#include <queue>


using namespace std;




void BFS(int node,list<int> rList,bool isVisited){


    list<int>::iterator i,iend;


    queue<int> q;


    isVisited[node]=true;


    q.push(node);


    while(!q.empty()){


        int a = q.front();


        q.pop();


        iend = rList[a].end();


        for(i=rList[a].begin();i!=iend;i++){


            if(!isVisited[i]){


                isVisited[i]=true;


                q.push(i);


            }


        }


    }


}
