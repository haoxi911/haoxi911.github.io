Depth First Search(DFS) – Use Recursion

Oct 24, 2010

This post implements a simple DFS based on recursive theory, the following code snippet accept a graph as input, and validate whether it is a connected graph.



#include <iostream>


#include <list>


using namespace std;




bool DFS(int,list<int>,bool);


void setFlag(int,list<int>,bool,int);




bool DFS(int node,list<int> rList,bool isVisited){


    int lastNode = new int[node+1];


    for(int j=0;j<=node;j++)


        lastNode[j]=0;


    for(int j=1;j<=node;j++)


        isVisited[j]=false;


    cout<<“DFS path is :”<<endl;


    setFlag(1,rList,isVisited,lastNode);


}


void setFlag(int node,list<int> rList,bool isVisited,int lastNode){


    bool haveNode = false;


    list<int>::iterator i,iend;


    isVisited[node] = true;


    if(lastNode[node]!=0)


        cout<<lastNode[node]<<“-“<<node<<endl;


    iend = rList[node].end();


    for(i=rList[node].begin();i!=iend;i++){


        if(!isVisited[i]){


            lastNode[i] = node;


            setFlag(i,rList,isVisited,lastNode);


            haveNode = true;


        }


    }


    if(!haveNode){


        if(lastNode[node]==1){


            bool flag = false;


            iend = rList[1].end();


            for(i=rList[1].begin();i!=iend;i++){


                if(!isVisited[i])


                    flag = true;


            }


            if(!flag)    return;


        }


        setFlag(lastNode[node],rList,isVisited,lastNode);


    }


}




int main(){


    int node,edge,j,a,b,ct;


    list<int> rList;


    bool *isVisited;


    while( cin>>node&&node!=0){


        rList = new list<int>[node+1];


        isVisited = new bool[node+1];


        isVisited[0] = true;




        cin>>edge;


        for(j=0;j<edge;j++){


            cin>>a>>b;


            rList[a].push_back(b);


            rList[b].push_back(a);


        }


        DFS(node,rList,isVisited);


        ct = 0;


        for(j=1;j<=node;j++){


            if(isVisited[j]) ct++;


        }


        if(ct == node) cout<<“Is a Map”<<endl;


        else cout<<“Is not a Map”<<endl;


    }


    return 0;


}
